SELECT 
    table,
    formatReadableSize(sum(bytes_on_disk)) AS size
FROM system.parts
WHERE database = 'ubac' AND table = 'ClientLog'
GROUP BY table;


SELECT 
    database,
    formatReadableSize(sum(bytes_on_disk)) AS size
FROM system.parts
GROUP BY database
ORDER BY sum(bytes_on_disk) DESC;
